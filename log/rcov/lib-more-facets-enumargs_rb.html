 16 # This is a simple reimplementation of the core Enumerable module
 17 # to allow the methods to take and pass-on arbitrary arguments to the
 18 # underlying #each call. This library uses Enumerator and scans
 19 # Enumerable so it can alwasy stay in sync.
 20 #
 21 # NOTE Any Enumerable method with a negative arity cannot do pass arguments
 22 # due to ambiguity in the argument count. So the methods #inject and #zip
 23 # do NOT work this way, but simply work as they do in Enumerable.
 24 # The method #find (and #detect) though has been made to work by removing
 25 # its rarely used optional parameter and providing instead an optional
 26 # keyword parameter (:ifnone => ...). Please keep these difference in mind.
 27 #
 28 #   require 'enumargs'
 29 #
 30 #   class T
 31 #     include Enumerable::Arguments
 32 #     def initialize(arr)
 33 #       @arr = arr
 34 #     end
 35 #     def each(n)
 36 #       arr.each{ |e| yield(e+n) }
 37 #     end
 38 #   end
 39 #
 40 #   t = T.new([1,2,3])
 41 #   t.collect(4)
 42 #   #=> [5,6,7]
 43 #

 47   def self.wrap_enumerable_method( methodname )
 48 
 49     m = methodname
 50     meth = Enumerable.instance_method(m)
 51     arity = meth.arity
 52 
 53     case arity <=> 0
 54     when 0
 55       class_eval %{
 56         def #{m}( *args, &yld )
 57           enum_for(:each, *args).#{m}( &yld )
 58         end
 59       }
 60     when 1
 61       class_eval %{
 62         def #{m}( *args, &yld )
 63           args, each_args = args[0...#{arity}], args[#{arity}..-1]
 64           enum_for(:each, *each_args).#{m}( *args, &yld )
 65         end
 66       }

 80   # Make exception for #find (a negative arity method) to accept
 81   # keyword argument.

 83   #   ObjectSpace.find(Class, :ifnone=>lambda{1}) { |e| ... }
 84   #   ObjectSpace.find(Class, :ifnone=>lambda{1}) { |e| ... }

 86   def find(*args, &yld)  # future use **keys ?
 87     if Hash === args.last and args.last.key?(:ifnone)
 88       ifnone = args.last.delete(:ifnone)
 89       args.pop if args.last.empty?
 90       enum_for(:each, *args).find( ifnone, &yld )
 91     else
 92       enum_for(:each, *args).find( &yld )
 93     end
 94   end
